        printf("%d is smaller than %d", b, a);
    }
    else if (a = b) {
        printf("They are equal");
    }
    else {
        printf("%d is greater than %d", b, a);
        printf("%d is smaller than %d", a, b);
    }
    return 0;
}

Да се напише програма, която чете час и минути от 24-часово денонощие, въведени от потребителя, и изчислява колко ще е часът след 15 минути. Резултатът да се отпечата във формат hh:mm. Часовете могат да са между 0 и 23, а минутите могат да са между 0 и 59. Часовете се изписват с една или две цифри. Минутите се изписват винаги с по две цифри, с водеща нула когато е необходимо. 

#include <stdio.h>

int main(void) {
    int hours;
    scanf("%d", &hours);
    int minutes;
    scanf("%d", &minutes);
    minutes += 15;
    if (minutes >= 60) {
        hours += 1;
        minutes -= 60;
    }
    printf("%02d:%02d", hours, minutes);
    return 0;
}

Дадено е цяло число – брой точки. Върху него се начисляват бонус точки по следните правила:
Ако числото е до 100 включително, бонус точките са 5.
Ако числото е по-голямо от 100, бонус точките са 20% от числото.
Ако числото е по-голямо от 1000, бонус точките са 10% от числото.
Допълнителни бонус точки (начисляват се отделно от предходните):
Ако числото е четно, се начисляват 2 допълнителни точки.
Ако числото завършва на 5, се начисляват 5 допълнителни точки.

#include <stdio.h>

int main(void) {
    int points;
    float bonus;
    scanf("%d", &points);
    if (1000 < points) {
        bonus += 0.1*points;
    }
    else if (100 < points) {
        bonus += 0.2*points;
    }
    else if (0 < points && points < 100) {
        bonus += 5;
    }

    if (points % 2 == 0) {
        bonus += 1;
    }

    if (points % 10 == 5) {
        bonus += 2;
    }

    printf("%lf\n", bonus);
    printf("%lf\n", points + bonus);
    return 0;
}

Трима спортисти състезатели финишират за някакъв брой секунди (между 1 и 50). Да се напише програма, която чете времената на състезателите, въведени от потребителя, и пресмята средното им време във формат "минути:секунди". Секундите да се извеждат с водеща нула (2 "02", 7 "07", 35 "35").

#include <stdio.h>

int main(void) {
    int first;
    scanf("%d",&first);
    int second;
    scanf("%d",&second);
    int third;
    scanf("%d",&third);
    int seconds = first + second + third;
    int minutes;
    while (seconds > 60) {
        minutes += 1;
        seconds -= 60;
    }

    printf("%d:%02d",minutes, seconds);
    return 0;
}
