#include <stdio.h>
int count(int num) {
        int count = 0;
        while (num != 0) {
                num = num / 10;
                count++;
        }
        return count;
}
int main(void) {
        int number;
        scanf("%d", &number);
        printf("%d", count(number));
        return 0;

#include <stdio.h>

void binary(int num) {
        int number = num;
        int c = 0;
        while (num != 0) {
                c++;
                num = num / 2;
        }
        int bnumber[c];
        int length = c;
        while (number != 0) {
                bnumber[c-1] = number % 2;
                number = number / 2;
                c--;
        }

        for (int i = 0; i < length; i++) {
                printf("%d", bnumber[i]);
        }
}
int main(void) {
        int number;
        scanf("%d", &number);
        binary(number);
        return 0;
}

Да се състави програма, чрез която се въвеждат 7 цели числа от интервала [-5000..5000] в едномерен масив. Програмата да изведе числото, което е най-близко до средната стойност на въведените числа и къде се намира.
Пример: 1,2,3,4,5,6,7
Изход: средна стойност 4, най-близка стойност 4 на 4-то място


#include <stdio.h>

int main(void) {
    int arr[7] = { };
    int num;
    for (int i = 0; i < 7; i++) {
        scanf("%d", &num);
        if (-5000 <= num && num <= 5000) {
            arr[i] = num;
        }
        else {
            printf("Wrong input\n");
            i--;
        }
    }
    int sum = 0;
    for (int i = 0; i < 7; i++) {
        sum += arr[i];
    }
    int average = sum / 7;
    int max = 0;
    for (int i = 0; i < 7; i++) {
        if(max<arr[i]){
            max = arr[i];
        }
    }
    int diff = max;
    int *pd;
    int index;
    for (int i = 0; i < 7; i++) {
        if (arr[i] - average == 0) {
            pd = &arr[i];
            index = i;
            break;
        }
        else if (arr[i]-average <= diff) {
            if (arr[i]-average < 0) {
                diff = (average-arr[i])* (-1);
                pd = &arr[i];
                index = i;
            }
            else {
                diff = average-arr[i];
                pd = &arr[i];
                index = i;
            }
        }
    }
    printf("number: %d\n", *pd);
    printf("place: %d\n", index + 1);
    return 0;
}
