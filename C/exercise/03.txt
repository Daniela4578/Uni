Да се прочете от клавиатурата цяло число и да се изведе дали;
- е четно или нечетно
- е положително, отрицателно или нула
- се дели едновременно на З и 5

#include<stdio.h>

int main() {
    int num ;
    scanf("%d",&num);
    if (num % 2 == 0) {
        printf("%d is even\n", num);
    }
    else {
        printf("%d is odd\n", num);
    }

    if (num < 0) {
        printf("%d is negative\n", num); 
    }
    else if (num == 0) {
        printf("%d is zero\n", num);
    }
    else {
        printf("%d is positive\n", num);
    }

    if (num % 3 == 0 && num % 5 == 0) {
        printf("Yes\n");
    }
    else {
        printf("No\n");
    }


    return 0;
}

- ако могат изведете вида на триъгълника: равностранен, равнобедрен или разностранен.

int main(void) {
    int a;
    int b;
    int c;
    scanf("%d",&a);
    scanf("%d",&b);
    scanf("%d",&c);
    if (a+b > c && b+c > a && c+a > b) {
        printf("Triangle!");
	if (a==b && b==c && c==a) {
            printf("Equilateral triangle");
        }
        else if (a == b || b == c || c == a) {
            printf("Isosceles triangle");
        }
        else {
            printf("Scalene triangle");
        }
    }
    else {
        printf("Not a triangle!");
    }

Да се въведат числа, докато потребителят не въведе 0. След това програмната трябва да изведе брой въведени числа, тяхната сума и средноаритметичното им.

#include <stdio.h>

int main(void) {
        int num;
        scanf("%d",&num);
        int sum = 0;
        int count = 0;
        while(num!=0) {
                sum += num;
                count++;
                scanf("%d",&num);
        }
        printf("Sum:%d\n",sum);
        printf("Count:%d\n",count);
        printf("Average:%d\n",sum/count);
        return 0;
}
